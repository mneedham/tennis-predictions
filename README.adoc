  -v $PWD/data:/data \

[source, bash]
----
docker run \
  -v $PWD/data:/data \
  -v $PWD/dumps:/dumps \
  --entrypoint bin/neo4j-admin \
  neo4j:4.4.7 dump \
  --database=neo4j \
  --to=/dumps/backup-`date '+%s'`.dump
----
